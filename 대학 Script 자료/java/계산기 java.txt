package kr.ac.a2023374774;

import android.os.Bundle;

import com.google.android.material.snackbar.Snackbar;

import androidx.appcompat.app.AppCompatActivity;

import android.view.Gravity;
import android.view.View;

import androidx.navigation.NavController;
import androidx.navigation.Navigation;
import androidx.navigation.ui.AppBarConfiguration;
import androidx.navigation.ui.NavigationUI;

import kr.ac.a2023374774.databinding.ActivityMainBinding;

import android.view.Menu;
import android.view.MenuItem;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {


    TextView vew;
    EditText edt1, edt2;
     Button btn1, btn2, btn3, btn4;
     String s1, s2, ds;
     Integer var1, var2, sum;

     Double dVar1, dVar2, dSum;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);

        setContentView(R.layout.activity_main);

        btn1 = findViewById(R.id.Btn1);
        btn2 = findViewById(R.id.Btn2);
        btn3 = findViewById(R.id.Btn3);
        btn4 = findViewById(R.id.Btn4);

        edt1 = findViewById(R.id.Edt1);
        edt2 = findViewById(R.id.Edt2);

        vew = findViewById(R.id.Vew);

        btn1.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                s1 = edt1.getText().toString().trim();
                s2 = edt2.getText().toString().trim();
                if(!s1.isEmpty() && !s2.isEmpty()){
                    var1 = Integer.parseInt(s1);
                    var2 = Integer.parseInt(s2);

                    sum = var1 + var2;
                    vew.setText("값:"+sum.toString());

                }else{
                    vew.setText("값이 비었습니다.");
                }
            }
        });

        btn2.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                s1 = edt1.getText().toString().trim();
                s2 = edt2.getText().toString().trim();
                if(!s1.isEmpty() && !s2.isEmpty()){
                    var1 = Integer.parseInt(s1);
                    var2 = Integer.parseInt(s2);

                    sum = var1 - var2;
                    vew.setText("값:"+sum.toString());

                }else{
                    vew.setText("값이 비었습니다.");
                }
            }
        });

        btn3.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                s1 = edt1.getText().toString().trim();
                s2 = edt2.getText().toString().trim();
                if(!s1.isEmpty() && !s2.isEmpty()){
                    var1 = Integer.parseInt(s1);
                    var2 = Integer.parseInt(s2);

                    sum = var1 * var2;
                    vew.setText("값:"+sum.toString());

                }else{
                    vew.setText("값이 비었습니다.");
                }
            }
        });

        btn4.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                s1 = edt1.getText().toString().trim();
                s2 = edt2.getText().toString().trim();
                if(!s1.isEmpty() && !s2.isEmpty()){
                    dVar1 = Double.parseDouble(s1);
                    dVar2= Double.parseDouble(s2);

                    dSum = dVar1 / dVar2;
                    ds = String.format("%.1f",dSum);
                    vew.setText("값:"+ds);

                }else{
                    vew.setText("값이 비었습니다.");
                }
            }
        });

        edt1.setOnFocusChangeListener((v,hasfocus)->{
            if(hasfocus){
                edt1.setHint("");
                edt1.setGravity(Gravity.START);
            }
            else if(edt1.getText().toString().trim().isEmpty()) {
                edt1.setHint("첫번째 숫자");
                edt1.setGravity(Gravity.CENTER);
            }
        });

        edt2.setOnFocusChangeListener((v,hasfocus)->{
            if(hasfocus){
                edt2.setHint("");
                edt2.setGravity(Gravity.START);
            }
            else if(edt2.getText().toString().trim().isEmpty()) {
                edt2.setHint("두번째 숫자");
                edt2.setGravity(Gravity.CENTER);
            }
        });

    }
}
